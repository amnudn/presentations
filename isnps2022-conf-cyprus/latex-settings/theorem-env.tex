% Adjusted thm-environment with bullet at end
\newcommand{\newmarkedtheorem}[1]{%
  \newenvironment{#1}
    {\pushQED{\leavevmode\unskip\penalty9999 \hbox{}\nobreak\hfill
  \quad\hbox{\(\bullet\)}}
    \csname inner@#1\endcsname}
    {\popQED\csname endinner@#1\endcsname}%
  \newtheorem{inner@#1}%
}
